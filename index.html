Method B: Manual WSL Setup
1. Install WSL Ubuntu
wsl --install
2. Build llama.cpp
sudo apt update && sudo apt install build-essential git cmake -y
git clone https://github.com/ggerganov/llama.cpp
cd llama.cpp && mkdir build && cd build
cmake .. && cmake --build . --config Release
3. Download & Merge Model
mkdir -p ~/models
# Download both parts
curl -L -o ~/models/q4_km-00001-of-00002.gguf https://huggingface.co/Qwen/Qwen2.5-7B-Instruct-GGUF/resolve/main/qwen2.5-7b-instruct-q4_k_m-00001-of-00002.gguf
curl -L -o ~/models/q4_km-00002-of-00002.gguf https://huggingface.co/Qwen/Qwen2.5-7B-Instruct-GGUF/resolve/main/qwen2.5-7b-instruct-q4_k_m-00002-of-00002.gguf
# Merge using the built tool
./build/bin/llama-gguf-split --merge ~/models/q4_km-00001-of-00002.gguf ~/models/Qwen2.5-7B-Instruct.Q4_K_M.gguf
